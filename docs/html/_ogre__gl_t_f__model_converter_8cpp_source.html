21 Ogre::VertexBufferPackedVec Ogre_glTF_modelConverter::constructVertexBuffer(const std::vector<Ogre_glTF_vertexBufferPart>& parts) const
22 {
23  Ogre::VertexElement2Vec vertexElements;
24  size_t stride{ 0 }, strideInElements{ 0 };
25  size_t vertexCount{ 0 }, previousVertexCount{ 0 };
26  for (const auto& part : parts)
27  {
28  vertexElements.emplace_back(part.type, part.semantic);
29  strideInElements += part.perVertex;
30  stride += part.buffer->elementSize() * part.perVertex;
31  vertexCount = part.vertexCount;
32 
33  //Sanity check
34  if (previousVertexCount != 0)
35  {
36  if (vertexCount == previousVertexCount)
37  {
38  previousVertexCount = vertexCount;
39  }
40  else
41  {
42  throw std::runtime_error("Part of vertex buffer for the same primitive have different vertex counts!");
43  }
44  }
45  else
46  {
47  previousVertexCount = vertexCount;
48  }
49  }
50 
51  OgreLog("There will be " + std::to_string(vertexCount) + " vertices with a stride of " + std::to_string(stride) + " bytes");
52 
53  Ogre_glTF_geometryBuffer<float> finalBuffer(vertexCount * strideInElements);
54  size_t bytesWrittenInCurrentStride;
55  for (size_t vertexIndex = 0; vertexIndex < vertexCount; ++vertexIndex)
56  {
57  bytesWrittenInCurrentStride = 0;
58  for (const auto& part : parts)
59  {
60  memcpy(finalBuffer.dataAddress() + (bytesWrittenInCurrentStride + vertexIndex * stride),
61  (part.buffer->dataAddress() + (vertexIndex * part.getPartStride())),
62  part.getPartStride()
63  );
64  bytesWrittenInCurrentStride += part.getPartStride();
65  }
66  }
67 
68  //OgreLog("Final content of the buffer:");
69  //finalBuffer._debugContentToLog();
70 
71  Ogre::VertexBufferPackedVec vec;
72  auto vertexBuffer = Ogre_glTF_modelConverter::getVaoManager()->
73  createVertexBuffer(vertexElements,
74  vertexCount,
75  Ogre::BT_IMMUTABLE,
76  finalBuffer.data(),
77  false);
78 
79  vec.push_back(vertexBuffer);
80  return vec;
81 }

175 Ogre::IndexBufferPacked* Ogre_glTF_modelConverter::extractIndexBuffer(int accessorID) const
176 {
177  OgreLog("Extracting index buffer");
178  const auto& accessor = model.accessors[accessorID];
179  const auto& bufferView = model.bufferViews[accessor.bufferView];
180  auto& buffer = model.buffers[bufferView.buffer];
181  const auto byteStride = accessor.ByteStride(bufferView);
182  const auto indexCount = accessor.count;
183  Ogre::IndexBufferPacked::IndexType type;
184 
185  if (byteStride < 0)
186  throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
187 
188  auto convertTo16Bit{ false };
189  switch (accessor.componentType)
190  {
191  default:
192  throw std::runtime_error("Unrecognized index data format");
193  case TINYGLTF_COMPONENT_TYPE_BYTE:
194  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_BYTE:
195  convertTo16Bit = true;
196  case TINYGLTF_COMPONENT_TYPE_SHORT:
197  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT:
198  {
199  type = Ogre::IndexBufferPacked::IT_16BIT;
200  auto geometryBuffer = Ogre_glTF_geometryBuffer<Ogre::uint16>(indexCount);
201  if (convertTo16Bit) loadIndexBuffer(geometryBuffer.data(), buffer.data.data(), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
202  else loadIndexBuffer(geometryBuffer.data(), reinterpret_cast<Ogre::uint16*>(buffer.data.data()), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
203  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geometryBuffer.dataAddress(), false);
204  }
205  case TINYGLTF_COMPONENT_TYPE_INT:;
206  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_INT:
207  {
208  type = Ogre::IndexBufferPacked::IT_32BIT;
209  auto geometryBuffer = Ogre_glTF_geometryBuffer<Ogre::uint32>(indexCount);
210  loadIndexBuffer(geometryBuffer.data(), reinterpret_cast<Ogre::uint32*>(buffer.data.data()), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
211  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geometryBuffer.dataAddress(), false);
212  }
213  }

238 Ogre_glTF_vertexBufferPart Ogre_glTF_modelConverter::extractVertexBuffer(const std::pair<std::string, int>& attribute) const
239 {
240  const auto elementScemantic = getVertexElementScemantic(attribute.first);
241  const auto& accessor = model.accessors[attribute.second];
242  const auto& bufferView = model.bufferViews[accessor.bufferView];
243  const auto& buffer = model.buffers[bufferView.buffer];
244  const auto vertexBufferByteLen = bufferView.byteLength;
245  const auto numberOfElementPerVertex = getVertexBufferElementsPerVertexCount(accessor.type);
246  const auto elementOffsetInBuffer = bufferView.byteOffset + accessor.byteOffset;
247  size_t bufferLenghtInBufferBasicType;
248 
249  std::unique_ptr<Ogre_glTF_geometryBuffer_base> geometryBuffer{ nullptr };
250 
251  switch (accessor.componentType)
252  {
253  case TINYGLTF_COMPONENT_TYPE_DOUBLE:
254  throw std::runtime_error("Double pressision not implemented!");
255  case TINYGLTF_COMPONENT_TYPE_FLOAT:
256  bufferLenghtInBufferBasicType = (vertexBufferByteLen / sizeof(float));
257  geometryBuffer = std::make_unique<Ogre_glTF_geometryBuffer<float>>(bufferLenghtInBufferBasicType);
258  break;
259 
260  default:
261  throw std::runtime_error("Unrecognized vertex buffer coponent type");
262  }
263 
264  Ogre::VertexElementType elementType{};
265  if (numberOfElementPerVertex == 2) elementType = Ogre::VET_FLOAT2;
266  if (numberOfElementPerVertex == 3) elementType = Ogre::VET_FLOAT3;
267  if (numberOfElementPerVertex == 4) elementType = Ogre::VET_FLOAT4;
268 
269  if (bufferView.byteStride == 0) OgreLog("Vertex buffer is 'tightly packed'");
270  const auto byteStride = accessor.ByteStride(bufferView);
271  if (byteStride < 0) throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
272  const auto vertexCount = accessor.count;
273 
274  const auto vertexElementLenghtInBytes = numberOfElementPerVertex * geometryBuffer->elementSize();
275  OgreLog("A vertex element on this buffer is " + std::to_string(vertexElementLenghtInBytes) + " bytes long");
276  for (size_t vertexIndex = 0; vertexIndex < vertexCount; vertexIndex++)
277  {
278  const size_t destOffset = vertexIndex * vertexElementLenghtInBytes;
279  const size_t sourceOffset = elementOffsetInBuffer + vertexIndex * byteStride;
280 
281  memcpy((geometryBuffer->dataAddress() + destOffset),
282  (buffer.data.data() + sourceOffset),
283  vertexElementLenghtInBytes);
284  }
285 
286  //geometryBuffer->_debugContentToLog();
287 
288  return {
289  std::move(geometryBuffer),
290  elementType,
291  elementScemantic,
292  vertexCount,
293  numberOfElementPerVertex
294  };
295 }
