18 Ogre::VertexBufferPackedVec Ogre_glTF_modelConverter::constructVertexBuffer(const std::vector<Ogre_glTF_vertexBufferPart>& parts) const
19 {
20  Ogre::VertexElement2Vec vertexElements;
21 
22  size_t stride { 0 }, strideInElements { 0 };
23  size_t vertexCount { 0 }, previousVertexCount { 0 };
24 
25  for(const auto& part : parts)
26  {
27  vertexElements.emplace_back(part.type, part.semantic);
28  strideInElements += part.perVertex;
29  stride += part.buffer->elementSize() * part.perVertex;
30  vertexCount = part.vertexCount;
31 
32  //Sanity check
33  if(previousVertexCount != 0)
34  {
35  if(vertexCount != previousVertexCount)
36  throw std::runtime_error("Part of vertex buffer for the same primitive have different vertex counts!");
37  }
38  else
39  previousVertexCount = vertexCount;
40  }
41 
42  OgreLog("There will be " + std::to_string(vertexCount) + " vertices with a stride of " + std::to_string(stride) + " bytes");
43 
44  Ogre_glTF_geometryBuffer<float> finalBuffer(vertexCount * strideInElements);
45  size_t bytesWrittenInCurrentStride { 0 };
46  for(size_t vertexIndex = 0; vertexIndex < vertexCount; ++vertexIndex)
47  {
48  bytesWrittenInCurrentStride = 0;
49  for(const auto& part : parts)
50  {
51  memcpy(finalBuffer.dataAddress() + (bytesWrittenInCurrentStride + vertexIndex * stride),
52  (part.buffer->dataAddress() + (vertexIndex * part.getPartStride())),
53  part.getPartStride());
54  bytesWrittenInCurrentStride += part.getPartStride();
55  }
56  }
57 
58  Ogre::VertexBufferPackedVec vec;
59  auto vertexBuffer = getVaoManager()->createVertexBuffer(vertexElements,
60  vertexCount,
61  Ogre::BT_IMMUTABLE,
62  finalBuffer.data(),
63  false);
64 
65  vec.push_back(vertexBuffer);
66  return vec;
67 }

175

191  subMesh->_compileBoneAssignments();
192  }
193  }
194

196  OgreMesh->_setBoundingSphereRadius(boundingBox.getRadius());

198 
199  return OgreMesh;

235 Ogre::IndexBufferPacked* Ogre_glTF_modelConverter::extractIndexBuffer(int accessorID) const
236 {
237  OgreLog("Extracting index buffer");
238  const auto& accessor = model.accessors[accessorID];
239  const auto& bufferView = model.bufferViews[accessor.bufferView];
240  auto& buffer = model.buffers[bufferView.buffer];
241  const auto byteStride = accessor.ByteStride(bufferView);
242  const auto indexCount = accessor.count;
243  Ogre::IndexBufferPacked::IndexType type;
244 
245  if(byteStride < 0)
246  throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
247 
248  auto convertTo16Bit { false };
249  switch(accessor.componentType)
250  {
251  default:
252  throw std::runtime_error("Unrecognized index data format");
253  case TINYGLTF_COMPONENT_TYPE_BYTE:
254  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_BYTE:
255  convertTo16Bit = true;
256  case TINYGLTF_COMPONENT_TYPE_SHORT:
257  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT:
258  {
259  type = Ogre::IndexBufferPacked::IT_16BIT;
260  auto geometryBuffer = Ogre_glTF_geometryBuffer<Ogre::uint16>(indexCount);
261  if(convertTo16Bit)
262  loadIndexBuffer(geometryBuffer.data(), buffer.data.data(), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
263  else
264  loadIndexBuffer(geometryBuffer.data(), reinterpret_cast<Ogre::uint16*>(buffer.data.data()), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
265  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geometryBuffer.dataAddress(), false);
266  }
267  case TINYGLTF_COMPONENT_TYPE_INT:;
268  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_INT:
269  {
270  type = Ogre::IndexBufferPacked::IT_32BIT;
271  auto geometryBuffer = Ogre_glTF_geometryBuffer<Ogre::uint32>(indexCount);
272  loadIndexBuffer(geometryBuffer.data(), reinterpret_cast<Ogre::uint32*>(buffer.data.data()), indexCount, bufferView.byteOffset + accessor.byteOffset, byteStride);
273  return getVaoManager()->createIndexBuffer(type, indexCount, Ogre::BT_IMMUTABLE, geometryBuffer.dataAddress(), false);
274  }
275  }
276 }

302 Ogre_glTF_vertexBufferPart Ogre_glTF_modelConverter::extractVertexBuffer(const std::pair<std::string, int>& attribute, Ogre::Aabb& boundingBox) const
303 {
304  const auto elementScemantic = getVertexElementScemantic(attribute.first);
305 
306  if(elementScemantic == Ogre::VES_BLEND_INDICES)
307  {
308  OgreLog("Blend Indices...");
309  }
310 
311  if(elementScemantic == Ogre::VES_BLEND_WEIGHTS)
312  {
313  OgreLog("Blend weights");
314  }
315 
316  const auto& accessor = model.accessors[attribute.second];
317  const auto& bufferView = model.bufferViews[accessor.bufferView];
318  const auto& buffer = model.buffers[bufferView.buffer];
319  const auto vertexBufferByteLen = bufferView.byteLength;
320  const auto numberOfElementPerVertex = getVertexBufferElementsPerVertexCount(accessor.type);
321  const auto elementOffsetInBuffer = bufferView.byteOffset + accessor.byteOffset;
322  size_t bufferLenghtInBufferBasicType { 0 };
323 
324  std::unique_ptr<Ogre_glTF_geometryBuffer_base> geometryBuffer { nullptr };
325  Ogre::VertexElementType elementType {};
326 
327  switch(accessor.componentType)
328  {
329  case TINYGLTF_COMPONENT_TYPE_DOUBLE:
330  throw std::runtime_error("Double pressision not implemented!");
331  case TINYGLTF_COMPONENT_TYPE_FLOAT:
332  bufferLenghtInBufferBasicType = (vertexBufferByteLen / sizeof(float));
333  geometryBuffer = std::make_unique<Ogre_glTF_geometryBuffer<float>>(bufferLenghtInBufferBasicType);
334  if(numberOfElementPerVertex == 2) elementType = Ogre::VET_FLOAT2;
335  if(numberOfElementPerVertex == 3) elementType = Ogre::VET_FLOAT3;
336  if(numberOfElementPerVertex == 4) elementType = Ogre::VET_FLOAT4;
337  break;
338  case TINYGLTF_COMPONENT_TYPE_UNSIGNED_SHORT:
339  bufferLenghtInBufferBasicType = (vertexBufferByteLen / sizeof(unsigned short));
340  geometryBuffer = std::make_unique<Ogre_glTF_geometryBuffer<unsigned short>>(bufferLenghtInBufferBasicType);
341  if(numberOfElementPerVertex == 2) elementType = Ogre::VET_USHORT2;
342  if(numberOfElementPerVertex == 4) elementType = Ogre::VET_USHORT4;
343  break;
344  default:
345  throw std::runtime_error("Unrecognized vertex buffer coponent type");
346  }
347 
348  if(bufferView.byteStride == 0) OgreLog("Vertex buffer is 'tightly packed'");
349  const auto byteStride = accessor.ByteStride(bufferView);
350  if(byteStride < 0) throw std::runtime_error("Can't get valid bytestride from accessor and bufferview. Loading data not possible");
351  const auto vertexCount = accessor.count;
352 
353  const auto vertexElementLenghtInBytes = numberOfElementPerVertex * geometryBuffer->elementSize();
354  OgreLog("A vertex element on this buffer is " + std::to_string(vertexElementLenghtInBytes) + " bytes long");
355  for(size_t vertexIndex = 0; vertexIndex < vertexCount; vertexIndex++)
356  {
357  const auto destOffset = vertexIndex * vertexElementLenghtInBytes;
358  const auto sourceOffset = elementOffsetInBuffer + vertexIndex * byteStride;
359 
360  memcpy((geometryBuffer->dataAddress() + destOffset),
361  (buffer.data.data() + sourceOffset),
362  vertexElementLenghtInBytes);
363  }
364 
365  if(elementScemantic == Ogre::VES_BLEND_INDICES)
366  {
367  volatile unsigned short* debugbuffer = reinterpret_cast<unsigned short*>(geometryBuffer->dataAddress());
368  for(size_t i = 0; i < (vertexElementLenghtInBytes * vertexCount) / sizeof(unsigned short); ++i)
369  OgreLog(std::to_string(debugbuffer[i]));
370  OgreLog("Done");
371  }
372 
373  /*
374  Update the bounding sizes once, when vertex positions has been read.
375  */
376  if(elementScemantic == Ogre::VES_POSITION)
377  {
378  Ogre::Vector3 minBounds, maxBounds;
379 
380  OgreLog("Updating bounding box size: ");
381 
382  minBounds.x = accessor.minValues.at(0);
383  minBounds.y = accessor.minValues.at(1);
384  minBounds.z = accessor.minValues.at(2);
385  OgreLog("Setting Min size: " + std::to_string(minBounds.x) + " " + std::to_string(minBounds.y) + " " + std::to_string(minBounds.z));
386 
387  maxBounds.x = accessor.maxValues.at(0);
388  maxBounds.y = accessor.maxValues.at(1);
389  maxBounds.z = accessor.maxValues.at(2);
390  OgreLog("Setting Max size: " + std::to_string(maxBounds.x) + " " + std::to_string(maxBounds.y) + " " + std::to_string(maxBounds.z));
391 
392  boundingBox.setExtents(minBounds, maxBounds);
393  }
394 
395  //geometryBuffer->_debugContentToLog();
396 
397  return {
398  std::move(geometryBuffer),
399  elementType,
400  elementScemantic,
401  vertexCount,
402  numberOfElementPerVertex
403  };
404 }
