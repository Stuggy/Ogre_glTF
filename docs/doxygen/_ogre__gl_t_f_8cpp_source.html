97  FileType detectType(const std::string& path) const
98  {
99  //Quickly open the file as binary and chekc if there's the gltf binary magic number
100  {
101  auto probe = std::ifstream(path, std::ios_base::binary);
102  if (!probe)
103  throw std::runtime_error("Could not open " + path);
104 
105  std::array<char, 5> buffer;
106  for (size_t i{ 0 }; i < 4; ++i)
107  probe >> buffer[i];
108  buffer[4] = 0;
109 
110  if (std::string("glTF") == std::string(buffer.data()))
111  {
112  OgreLog("Detected binary file thanks to the magic number at the start!");
113  return FileType::Binary;
114  }
115  }
116 
117  //If we don't have any better, check the file extension.
118  auto extension = path.substr(path.find_last_of('.') + 1);
119  std::transform(std::begin(extension), std::end(extension), std::begin(extension), [](char c) {return char(::tolower(int(c))); });
120  if (extension == "gltf") return FileType::Ascii;
121  if (extension == "glb") return FileType::Binary;
122 
123  return FileType::Unknown;
124  }

127  bool loadInto(Ogre_glTF_adapter& adapter, const std::string& path)
128  {
129  switch (detectType(path))
130  {
131  default:
132  case FileType::Unknown:

134  case FileType::Ascii:

137  case FileType::Binary:
