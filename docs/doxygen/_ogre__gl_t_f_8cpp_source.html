124  FileType detectType(const std::string& path) const
125  {
126  //Quickly open the file as binary and chekc if there's the gltf binary magic number
127  {
128  auto probe = std::ifstream(path, std::ios_base::binary);
129  if(!probe)
130  throw std::runtime_error("Could not open " + path);
131 
132  std::array<char, 5> buffer {};
133  for(size_t i { 0 }; i < 4; ++i)
134  probe >> buffer[i];
135  buffer[4] = 0;
136 
137  if(std::string("glTF") == std::string(buffer.data()))
138  {
139  OgreLog("Detected binary file thanks to the magic number at the start!");
140  return FileType::Binary;
141  }
142  }
143 
144  //If we don't have any better, check the file extension.
145  auto extension = path.substr(path.find_last_of('.') + 1);
146  std::transform(std::begin(extension), std::end(extension), std::begin(extension), [](char c) { return char(::tolower(int(c))); });
147  if(extension == "gltf") return FileType::Ascii;
148  if(extension == "glb") return FileType::Binary;
149 
150  return FileType::Unknown;
151  }

154  bool loadInto(Ogre_glTF_adapter& adapter, const std::string& path)
155  {
156  switch(detectType(path))
157  {
158  default:
159  case FileType::Unknown:
160  return false;
161  case FileType::Ascii:
162  OgreLog("Detected ascii file type");
163  return loader.LoadASCIIFromFile(&adapter.pimpl->model, &adapter.pimpl->error, path);
164  case FileType::Binary:
165  OgreLog("Deteted binary file type");
166  return loader.LoadBinaryFromFile(&adapter.pimpl->model, &adapter.pimpl->error, path);
167  }
168  }
